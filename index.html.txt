<script>
const API_URL = "/api/users";

async function fetchUsers() {
    const res = await fetch(API_URL);
    const users = await res.json();
    const searchVal = document.getElementById("search").value.toLowerCase();
    const tbody = document.querySelector("#userTable tbody");
    tbody.innerHTML = "";

    users.forEach(u => {
        if (searchVal && !(
            u.imie.toLowerCase().includes(searchVal) ||
            u.nazwisko.toLowerCase().includes(searchVal) ||
            u.pesel.toLowerCase().includes(searchVal)
        )) return;

        const tr = document.createElement("tr");
        tr.innerHTML = `
            <td>${u.imie}</td>
            <td>${u.nazwisko}</td>
            <td>${u.pesel}</td>
            <td>${u.status}</td>
            <td>${u.mandaty.join(", ")}</td>
            <td>${u.sprawy.join(", ")}</td>
        `;
        tbody.appendChild(tr);
    });
}

fetchUsers();
</script>
